create table Artist
(
  Artist_id int primary key,
  Name varchar (50)
);

create table Albums
(
  Albums_id int primary key,
  Album_name varchar (50),
  Years int not null
);

create table Tracks
(
  Track_id int primary key,
  Albums_id int references Albums(Albums_id),
  Title varchar (50),
  Times text not null
);

create table Genre 
(
  Genre_id int primary key,
  Genre_name varchar(50)
);

create table Artist_Genre 
(
  Artist_id int not null references Artist(Artist_id),
  Genre_id int not null references Genre(Genre_id),
  constraint pk primary key (Artist_id,Genre_id_id)
);

create table Artist_Albums
(
  Artist_id int not null references Artist(Artist_id),
  Albums_id int not null references Albums(Albums_id),
  constraint pk primary key (Artist_id,Albums_id)
);

create table Tracks_Collection
(
  Tracks_id int not null references Tracks(Tracks_id),
  Collection_id int not null references Collection(Collection_id),
  constraint pk primary key (Tracks_id,Collection_id)
);

create table Collection
(
  Collection_id int primary key,
  Collection_name varchar(50),
  Years int not null
);
